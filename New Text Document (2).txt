-- ORDER TABLE
CREATE TABLE ORDER_TABLE (
  ORDER_ID     VARCHAR2(50) PRIMARY KEY,
  CUSTOMER_ID  VARCHAR2(50),
  PRODUCT_ID   VARCHAR2(50),
  QUANTITY     NUMBER,
  STATUS       VARCHAR2(20),
  CREATED_DATE DATE
);

-- INVENTORY TABLE
CREATE TABLE INVENTORY_TABLE (
  PRODUCT_ID    VARCHAR2(50) PRIMARY KEY,
  AVAILABLE_QTY NUMBER
);
